CREATE TABLE Proizvodjaci(
    PIB INT(9) PRIMARY KEY NOT NULL,
    Naziv VARCHAR(45) NOT NULL,
    Grad VARCHAR(35) NOT NULL,
    Ulica VARCHAR(55) NOT NULL,
    Broj VARCHAR(10) NOT NULL,
    AdresaSajta VARCHAR(45) NOT NULL
) ENGINE = INNODB

CREATE TABLE Modeli(
    ModelID INT PRIMARY KEY NOT NULL,
    Oznaka VARCHAR(45) NOT NULL,
    JacinaMotora INT(3) NOT NULL,
    BrojVrata INT(1) NOT NULL,
    Cena INT NOT NULL,
    PIB INT(9),
    FOREIGN KEY(PIB) REFERENCES Proizvodjaci(PIB) 
    ON UPDATE CASCADE 
    ON DELETE NO ACTION
) ENGINE = INNODB
