# SPAJANJE Imena i Prezimena

SELECT concat(ime, ' ' , prezime) 
AS ImePrezime 
From radnici

#Spojiti imena i prezimena radnika i sortirati ih po duzini opadajuÄ‡e i prikazati duzinu

SELECT concat(ime, ' ' , prezime) 
AS ImePrezime, length(concat(ime, prezime)) 
as duzina 
From radnici 
order by length(ImePrezime) DESC;
